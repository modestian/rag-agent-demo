## 4. 数据类型转换

## 6. 字符串操作
```python
text = "Python"

# 索引与切片
text[0]         # 'P'
text[-1]        # 'n'
text[0:3]       # 'Pyt'
text[:3]        # 'Pyt'
text[3:]        # 'hon'

# 常用方法
len(text)                # 长度 → 6
text.upper()             # 大写 → 'PYTHON'
text.lower()             # 小写 → 'python'
text.startswith("Py")    # True
text.replace("Py", "My") # 'Mython'
"Py" in text             # True

# 格式化
name = "小明"
age = 20
f"我叫{name}，今年{age}岁"  # f-string (推荐)
"我叫{}，今年{}岁".format(name, age)
```

## 7. 列表 List
```python
# 创建列表
fruits = ["apple", "banana", "cherry"]
numbers = [1, 2, 3, 4, 5]

# 基本操作
fruits[0]               # 访问 → 'apple'
fruits[-1]              # 访问最后一个 → 'cherry'
fruits[0:2]             # 切片 → ['apple', 'banana']
fruits.append("orange") # 添加元素
fruits.insert(1, "grape") # 插入元素
fruits.remove("banana") # 删除元素
fruits.pop()            # 删除并返回最后一个

# 列表方法
len(fruits)             # 长度
sorted(fruits)          # 排序（返回新列表）
fruits.sort()           # 原地排序
fruits.reverse()        # 反转
```

## 9. 字典 Dictionary
```python

    "name": "张三",
    "age": 25,
    "city": "北京"

person.values()  # 所有值
person.items()   # 所有键值对
```

## 10. 集合 Set
```python

```

## 11. 条件语句
```python

    print("正数")

    print("零")

    print("负数")
```

## 12. 循环语句
```python
# for循环
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(fruit)

# 带索引的for循环
for i, fruit in enumerate(fruits):
    print(i, fruit)

# while循环
count = 0
while count < 5:
    print(count)
    count += 1

# 循环控制
for i in range(10):
    if i == 3:
        continue  # 跳过本次循环
    if i == 7:
        break     # 终止循环
    print(i)
```

## 13. 函数
```python
# 定义函数
def greet(name):
    """打招呼的函数"""
    return f"Hello, {name}!"

# 调用函数
result = greet("Alice")

# 默认参数
def power(base, exponent=2):
    return base ** exponent

# 可变参数
def sum_all(*args):
    return sum(args)

# 关键字参数
def person_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

# lambda函数
square = lambda x: x ** 2
```

## 14. 异常处理
```python
try:
    result = 10 / 0
except ZeroDivisionError:
    print("不能除以零")
except Exception as e:
    print(f"发生错误: {e}")
else:
    print("没有错误发生")
finally:
    print("无论是否出错都会执行")
```

## 15. 文件操作
```python

with open("example.txt", "w", encoding="utf-8") as f:
    f.write("Hello, Python!\n")

# 读取文件
with open("example.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)

# 逐行读取
with open("example.txt", "r", encoding="utf-8") as f:
```

## 17. 类与对象
```python
# 定义类
class Person:
    # 类属性
    species = "人类"

    # 构造方法
    def __init__(self, name, age):
        self.name = name  # 实例属性
        self.age = age

    # 实例方法
    def introduce(self):
        return f"我叫{self.name}，今年{self.age}岁"

    # 类方法
    @classmethod
    def get_species(cls):
        return cls.species

# 创建对象
p1 = Person("张三", 25)
print(p1.introduce())

# 继承
class Student(Person):
    def __init__(self, name, age, student_id):
        super().__init__(name, age)
        self.student_id = student_id
```

## 18. 列表推导式
```python
# 创建列表
squares = [x**2 for x in range(10)]  # [0, 1, 4, ..., 81]

# 带条件的推导式
even_squares = [x**2 for x in range(10) if x % 2 == 0]

# 字典推导式
square_dict = {x: x**2 for x in range(5)}
```
